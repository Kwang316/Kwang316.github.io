DVH Computation

# dvh_compute.py
import pydicom, numpy as np

def compute_dvh(dose_dcm_path, structure_mask: np.ndarray):
    ds    = pydicom.dcmread(dose_dcm_path)
    dose  = ds.pixel_array * float(ds.DoseGridScaling)  # → Gy
    dose_in_struct = dose[structure_mask > 0]

    bins      = np.linspace(0, dose.max(), 200)
    hist, _   = np.histogram(dose_in_struct, bins=bins)
    cum_dvh   = 1 - np.cumsum(hist) / hist.sum()        # cumulative DVH
    return bins[:-1].tolist(), cum_dvh.tolist()

# Key constraints checked:  D2% parotid ≤ 26 Gy, Dmax cord ≤ 45 Gy
