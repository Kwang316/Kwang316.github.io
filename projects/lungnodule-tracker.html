# nodule_tracker.py — core volume comparison
import SimpleITK as sitk
import numpy as np

def compute_volume_mm3(label_image: sitk.Image) -> float:
    spacing  = label_image.GetSpacing()
    voxel_vol = spacing[0] * spacing[1] * spacing[2]
    arr       = sitk.GetArrayFromImage(label_image)
    return np.sum(arr > 0) * voxel_vol

def volume_doubling_time(v1, v2, days_apart) -> float:
    if v2 <= v1: return float('inf')
    return days_apart * np.log(2) / np.log(v2 / v1)

def fleischner_flag(vdt_days: float) -> str:
    if vdt_days < 400:  return "HIGH_RISK"
    if vdt_days < 600:  return "INTERMEDIATE"
    return "LOW_RISK"
