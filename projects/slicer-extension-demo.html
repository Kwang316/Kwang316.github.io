Module Entry Point

import slicer
from slicer.ScriptedLoadableModule import *

class DemoModuleWidget(ScriptedLoadableModuleWidget):
    def setup(self):
        ScriptedLoadableModuleWidget.setup(self)
        uiWidget = slicer.util.loadUI(self.resourcePath('UI/DemoModule.ui'))
        self.layout.addWidget(uiWidget)
        self.ui = slicer.util.childWidgetVariables(uiWidget)
        self.ui.applyButton.connect('clicked(bool)', self.onApply)

    def onApply(self):
        logic = DemoModuleLogic()
        logic.process(slicer.mrmlScene.GetFirstNodeByClass('vtkMRMLScalarVolumeNode'))
